animate関数を使用し、マウスオーバーしたら右のマージンを140pxにし、マウスが外れたら0になるようにしています。

$(function() {
      $('#slideR').hover(
      function(){
        $(this).animate({'marginRight':'140px'},500);
      },
      function () {
        $(this).animate({'marginRight':'0'},500);
      }
    );
});
